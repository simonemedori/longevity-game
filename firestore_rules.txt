rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper: verifica che l'utente sia un admin registrato nella collezione admins
    function isAdmin() {
      return request.auth != null
        && request.auth.token.email != null
        && exists(/databases/$(database)/documents/artifacts/simulatore-longevity-ca/admins/$(request.auth.token.email));
    }

    // 1. Aule di gioco
    //    - Lettura: qualsiasi utente autenticato (anonymous o Google) che conosce il gameId
    //    - Creazione/eliminazione stanza: solo admin
    //    - Aggiornamento dati di gioco (allocazioni, eventi, punteggi): qualsiasi autenticato
    match /artifacts/simulatore-longevity-ca/public/data/games/{gameId} {
      allow read: if request.auth != null;
      allow create, delete: if isAdmin();
      allow update: if request.auth != null;
    }

    // 2. Lista admin: leggibile solo da utenti autenticati, non modificabile da client
    match /artifacts/simulatore-longevity-ca/admins/{email} {
      allow read: if request.auth != null;
      allow write: if false;
    }

    // 3. Protezione globale: blocca tutto il resto
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
